009020,000032: 
009021,000033:                                                                                                  #     THE KEYBOARD AND DISPLAY SYSTEM PROGRAM OPERATES UNDER EXECUTIVE
009022,000034:                                                                                                  #  CONTROL AND PROCESSES INFORMATION EXCHANGED BETWEEN THE AGC AND THE
009023,000035:                                                                                                  #  COMPUTER OPERATOR.  THE INPUTS TO THE PROGRAM ARE FROM THE KEYBOARD,
009024,000036:                                                                                                  #  FROM INTERNAL PROGRAMS, AND FROM THE UPLINK.
009025,000037:                                                                                                  #     THE LANGUAGE OF COMMUNICATION WITH THE PROGRAM IS A PAIR OF WORDS
009026,000038:                                                                                                  #  KNOWN AS VERB AND NOUN.  EACH OF THESE IS REPRESENTED BY A 2 CHARACTER
009027,000039:                                                                                                  #  OCTAL NUMBER.  THE VERB CODE INDICATES WHAT ACTION IS TO BE TAKEN, 
009028,000040:                                                                                                  #  NOUN CODE INDICATES TO WHAT THIS ACTION IS APPLIED.  NOUNS USUALLY
009029,000041:                                                                                                  #  REFER TO A GROUP OF ERASABLE REGISTERS.
009030,000042: 
009031,000043:                                                                                                  #  VERBS ARE GROUPED INTO DISPLAYS, LOADS, MONITORS (DISPLAYS THAT ARE
009032,000044:                                                                                                  #  UPDATED ONCE PER SECOND), SPECIAL FUNCTIONS, AND EXTENDED VERBS(THESE
009033,000045:                                                                                                  #  ARE OUTSIDE OF THE DOMAIN OF PINBALL AND CAN BE FOUND UNDER LOG SECTION
009034,000046:                                                                                                  #  :EXTENDED VERBS:).
009035,000047:                                                                                                  #  A LIST OF VERBS AND NOUNS IS GIVEN IN LOG SECTION :ASSEMBLY AND 
009036,000048:                                                                                                  #  OPERATION INFORMATION:.
009037,000049: 
009038,000050: 
009039,000051:                                                                                                  #  CALLING SEQUENCES -
009040,000052: 
009041,000053:                                                                                                  #  KEYBOARD:
009042,000054:                                                                                                  #     EACH DEPRESSION OF A KEYBOARD BUTTON ACTIVATES INTERRUPT KEYRUPT1
009043,000055:                                                                                                  #  AND PLACES THE 5 BIT KEY CODE INTO CHANNEL 15.  KEYRUPT1 PLACES THE KEY
009044,000056:                                                                                                  #  CODE INTO MPAC, ENTERS AN EXECUTIVE REQUEST FOR THE KEYBOARD AND DISPLAY
009045,000057:                                                                                                  #  PROGRAM (AT :CHARIN:), AND EXECUTES A RESUME.
009046,000058: 
009047,000059:                                                                                                  #  UPLINK:
009048,000060:                                                                                                  #     EACH WORD RECEIVED BY THE UPLINK ACTIVATES INTERRUPT UPRUPT WHICH
009049,000061:                                                                                                  #  PLACES THE 5 BIT KEY CODE INTO MPAC, ENTERS AN EXECUTIVE REQUEST FOR THE
009050,000062:                                                                                                  #  KEYBOARD AND DISPLAY PROGRAM (AT:CHARIN:) AND EXECUTES A RESUME.
009051,000063: 
009052,000064:                                                                                                  #  INTERNAL PROGRAMS:
009053,000065:                                                                                                  #     INTERNAL PROGRAMS CALL PINBALL AT :NVSUB: WITH THE DESIRED VERB/NOUN
009054,000066:                                                                                                  #  CODE IN A (LOW 6 BITS FOR NOUN, NEXT 6 BITS FOR VERB).  DETAILS
009055,000067:                                                                                                  #  DESCRIBED ON REMARKS CARDS JUST BEFORE :NVSUB: AND :NVSUBWAIT: (SEE
009056,000068: 
009057,000069:                                                                                                  #  SYMBOL TABLE FOR PAGE NUMBERS).
009058,000070:                                                                                                  #     THERE IS AN INTERLOCK SYSTEM FOR INTERNAL PROGRAMS THAT MUST BE USED
009059,000071:                                                                                                  #  IN CONJUNCTION WITH :NVSUB:.  THIS IS CALLED GRAB/FREE.  DETAILS
009060,000072:                                                                                                  #  DESCRIBED ON REMARKS JUST BEFORE :GRABDSP:, :GRABWAIT: (SEE SYMBOL TABLE
009061,000073:                                                                                                  #  FOR PAGE NUMBERS).
